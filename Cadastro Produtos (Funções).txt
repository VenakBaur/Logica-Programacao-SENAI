programa
{

	inclua biblioteca Util --> u

	inteiro escolhaMenu = 0
	cadeia produtos[999], classificacao[999]
	inteiro contador = -1
	inteiro resposta = 0
	inteiro indice = 0, testeIndice
	cadeia testeProduto = "", testeClassificacao = "", enter = "", resposta1
	
	funcao inicio()
	{

		enquanto (resposta != 4) {

			u.aguarde(1500)

			limpa()

			menu()
			leia(escolhaMenu)

			se (escolhaMenu == 1) {

				inserir()
			
			} senao se (escolhaMenu == 2) {

				atualizar()
			
			} senao se (escolhaMenu == 3) {

				ordemIndice()
			
			} senao se (escolhaMenu == 4) {

				excluir()
			
			} senao se (escolhaMenu == 5) {
				
				encerrar()
				
			} senao {

				escreva("Opcao invalida")
		
			}
		} 

		escreva("\nObrigado pela preferencia!\n")
			
	}

	funcao menu() {

		escreva("MENU\n\n")
		
		escreva("1 - Inserir\n")
		escreva("2 - Atualizar\n")
		escreva("3 - Exibir\n")
		escreva("4 - Excluir\n")
		escreva("5 - Encerrar\n\n")
		escreva("O que deseja fazer? ")

	}

	funcao inserir() {

		contador++
																
		faca {
		
			escreva("Insira um produto: ")
			leia(testeProduto)

			escreva("Insira uma classificacao: ")
			leia(testeClassificacao)

			se (testeProduto == "") {

				escreva("Indice Invalido") 
				
			}

			se (testeClassificacao == "") {

				escreva("Classificacao Invalida")
				
			}
	
		} enquanto (testeProduto == "" ou testeClassificacao == "")

		produtos[contador] = testeProduto
		classificacao[contador] = testeClassificacao
		
	}

	funcao atualizar() {

		faca {
		
			escreva("Qual indice deseja atualizar? ")
			leia(indice)

			se (indice < 0 ou indice > contador) {

				escreva("Indice Invalido")
				
			}
	
		} enquanto (indice < 0 ou indice > contador)

		escreva(produtos[indice] + " (indice " + indice + ") = ")
		leia(produtos[indice])
		
	}
	
	funcao ordemIndice() {

		escreva("Deseja exibir os indices de forma crescente ou decrescente? (C/D) ")
		leia(resposta1)

		se (resposta1 == "C" ou resposta1 == "c") {

			para (inteiro i = 0; i <= contador; i++) {

				se (produtos[i] == "") {
	
				} senao {

					escreva("Produto: " + produtos[i] + ", indice " + i + "\nClassificacao: " + classificacao[i] + "\n\n")
					
				}
				
			}
			
		} senao {

			para (inteiro i = contador; i >= 0; i--) {

				se (produtos[i] == "") {
	
				} senao {

					escreva("Produto: " + produtos[i] + ", indice " + i + "\nClassificacao: " + classificacao[i] + "\n\n")
					
				}
			
			}
		
		}

		escreva("Pressione enter para prosseguir: ")
		leia(enter)

	}

	funcao excluir() {

		escreva("Qual indice voce quer excluir? ")
		leia(indice)

		produtos[indice] = ""

		classificacao[indice] = ""
			 
	}

	funcao encerrar() {

		resposta = 4
		
	}
}
