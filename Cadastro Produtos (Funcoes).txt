programa
{

	inclua biblioteca Util --> u

	inteiro escolhaMenu = 0
	cadeia produtos[999]
	inteiro contador = 0
	inteiro resposta = 0
	inteiro indice = 0
	cadeia testeProduto = ""
	
	funcao inicio()
	{

		enquanto (resposta != 4) {

			u.aguarde(1500)

			limpa()

			menu()
			leia(escolhaMenu)

			se (escolhaMenu == 1) {

				inserir()
			
			} senao se (escolhaMenu == 2) {

				exibir()
			
			} senao se (escolhaMenu == 3) {

				excluir()
			
			} senao se (escolhaMenu == 4) {

				encerrar()
				
			
			} senao {

			escreva("Opcao invalida")
		
			}
		} 

		escreva("\nObrigado pela preferencia!\n")
			
	}

	funcao menu() {

		escreva("Menu\n\n")
		
		escreva("1 - Inserir\n")
		escreva("2 - Exibir\n")
		escreva("3 - Encerrar\n\n")
		escreva("O que deseja fazer? ")

	}

	funcao inserir() {

		faca {
		
		escreva("Qual indice voce quer atualizar? ")
		leia(indice)

			se (indice < 0) {

				escreva("Indice Invalido")
				
			}
	
		} enquanto (indice < 0)

		faca {

			escreva("Nome do produto: ")
			leia(testeProduto)
		
		} enquanto (testeProduto == "")

		produtos[indice] = testeProduto

		contador++
		
	}

	funcao exibir() {

		u.aguarde(1500)

		limpa()

		escreva(produtos[indice])
			                           
		}

	funcao excluir() {

		escreva("Qual indice voce quer excluir? ")
		leia(indice)

		produtos[indice] = ""

		contador--
			 
	}

	funcao encerrar() {

		resposta = 4
		
	}
}
