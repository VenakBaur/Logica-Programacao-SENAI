programa
{

	inclua biblioteca Util --> u

	inteiro escolhaMenu = 0
	cadeia produtos[999]
	inteiro contador = 0
	inteiro resposta = 0
	inteiro indice = 0
	cadeia testeProduto = "", enter = ""
	
	funcao inicio()
	{

		enquanto (resposta != 4) {

			u.aguarde(1500)

			limpa()

			menu()
			leia(escolhaMenu)

			se (escolhaMenu == 1) {

				inserir()
			
			} senao se (escolhaMenu == 2) {

				exibir()
			
			} senao se (escolhaMenu == 3) {

				excluir()
			
			} senao se (escolhaMenu == 4) {

				encerrar()
				
			
			} senao {

				escreva("Opcao invalida")
		
			}
		} 

		escreva("\nObrigado pela preferencia!\n")
			
	}

	funcao menu() {

		escreva("MENU\n\n")
		
		escreva("1 - Inserir\n")
		escreva("2 - Exibir\n")
		escreva("3 - Excluir\n")
		escreva("4 - Encerrar\n\n")
		escreva("O que deseja fazer? ")

	}

	funcao inserir() {
																
		faca {
		
			escreva("Insira um produto: ")
			leia(testeProduto)

			se (testeProduto == "") {

				escreva("Indice Invalido")
				
			}
	
		} enquanto (testeProduto == "")

		produtos[contador] = testeProduto

		contador++
		
	}
	
	funcao exibir() {

		u.aguarde(1500)

		limpa()

		para (inteiro i = 0; i < contador; i++) {

			se (produtos[i] == "") {

				i++
				
			}

			escreva("Produto: " + produtos[i] + ", indice " + i + "\n")

		}

		escreva("Pressione enter para prosseguir: ")
		leia(enter)
		
	}

	funcao excluir() {

		escreva("Qual indice voce quer excluir? ")
		leia(indice)

		produtos[indice] = ""
			 
	}

	funcao encerrar() {

		resposta = 4
		
	}
}
