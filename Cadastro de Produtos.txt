programa
{

	inteiro escolhaMenu
	cadeia produtos[999]
	inteiro contador = 0
	caracter resposta = 's'
	
	funcao inicio()
	{
		
		

		faca {

			limpa()

			menu()
			
			escreva("Deseja continuar? (S/N) ")
			leia(resposta)

			} enquanto (resposta == 's' ou resposta == 'S')

		escreva("\nObrigado pela preferencia!\n")
			
		}

	funcao menu() {

		escreva("Menu\n\n")
		
		escreva("1 - Inserir\n")
		escreva("2 - Exibir\n")
		escreva("3 - Encerrar\n\n")
		leia(escolhaMenu)

		se (escolhaMenu == 1) {

			inserir()
			
		} senao se (escolhaMenu == 2) {

			exibir()
			
		} senao se (escolhaMenu == 3) {

			encerrar()

		}

	}

	funcao inserir() {

		

		faca {

		escreva("Nome do produto: ")
		leia(produtos[contador])

		se (produtos[contador] == "") {

			inserir()
			
		}
		
		} enquanto (produtos[contador] == "")

		contador++
		
	}

	funcao exibir() {

		inteiro exibirProduto = 0

		escreva("Deseja exibir qual produto? ")
		leia(exibirProduto)

		se (exibirProduto > contador ou exibirProduto < 0) {

			exibir()
			
		}

		escreva("Produto: " + produtos[exibirProduto])
		
	}	

	funcao encerrar() {

		resposta = 'n'
		
	}
}
